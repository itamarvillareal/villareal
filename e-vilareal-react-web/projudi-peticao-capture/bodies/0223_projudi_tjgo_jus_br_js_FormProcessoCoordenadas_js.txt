function adicionarCoordenada(){
	var objLatitude = document.getElementById('Latitude');
	var objLongitude = document.getElementById('Longitude');
	var objPonto = document.getElementById('Ponto');
	
	if(objPonto.value=="" || objPonto.value ==undefined){
		alert('Informe o Numero do Ponto da Coordenada');
		return;
	}
	
	if(isLatitudeValida(objLatitude) && isLongitudeValida(objLongitude)){
		var Ponto = parseInt($('#Ponto').val());
		var Latitude = $('#Latitude').val();
		var Longitude = $('#Longitude').val();
		CoordenadasProcesso[Ponto-1]= {"lat":Latitude,"lon":Longitude};
		prepararCoordenadas(CoordenadasProcesso);
			
		$('#Ponto').val("");
		$('#Latitude').val("");
		$('#Longitude').val("");
	}
}

function prepararCoordenadas(objCoordenadasProcesso){
	var stCoordenadas="";
	for(var i = 0; i<objCoordenadasProcesso.length; i++){		
		if (objCoordenadasProcesso[i]!=undefined){
			stCoordenadas+= (i+1) + " [" + objCoordenadasProcesso[i].lat + ", " + objCoordenadasProcesso[i].lon + "]"
			if (i < (objCoordenadasProcesso.length-1)){
				stCoordenadas+=", "
			}
		}		
	}	
	if (objCoordenadasProcesso.length>=3){
		calcularPerimetro( objCoordenadasProcesso);
	}
	//console.log(JSON.stringify(objCoordenadasProcesso));
	$('#Coordenada1').html(stCoordenadas);
	$('#Coordenada').val(JSON.stringify(objCoordenadasProcesso));
}

function editarPassoSelecionar(){
	var objCoordenada = $('#Coordenada').val();
    const CoordenadasProcesso = ['undefined', 'null', ''].includes(objCoordenada)
        ? []
        : JSON.parse(objCoordenada);
	prepararCoordenadas(CoordenadasProcesso);	
}

function isLongitudeValida(obj) {
	var longitude = obj.value;
    if (!longitude.match(/^[-]?\d+[\.]?\d*$/)) {
        alert('Longitude invalida');		
        return false;
    }    
    
    if(!( longitude>-180 && longitude<180)){
		alert('Longitude invalida');		
        return false;
	}
	return true;
}

function isLatitudeValida(obj) {
	
	latitude = obj.value;
    if (!latitude.match(/^[-]?\d+[\.]?\d*d*$/)) {
		alert('Latitude invalida');				
        return false;
    }    
    if(!(latitude>-90 && latitude<90)){
		alert('Latitude invalida');		 
        return false;
	}
	return true;
}

function calcularPerimetro(objCoordenadasProcesso){
	var valPerimetro = 0.0;
	
	for (var i = 0; i<(objCoordenadasProcesso.length-1); i++){
		valPerimetro+= distanceCoordenda(objCoordenadasProcesso[i].lat, objCoordenadasProcesso[i].lon,  objCoordenadasProcesso[i+1].lat, objCoordenadasProcesso[i+1].lon, "k");
	}
	valPerimetro+= distanceCoordenda(objCoordenadasProcesso[objCoordenadasProcesso.length-1].lat, objCoordenadasProcesso[objCoordenadasProcesso.length-1].lon,  objCoordenadasProcesso[0].lat, objCoordenadasProcesso[0].lon, "k");
	
	$('#Perimetro').val(valPerimetro.toFixed(2) + " km");
}

function distanceCoordenda(lat1, lon1, lat2, lon2, unit) {
  const radlat1 = Math.PI * lat1/180;
  const radlat2 = Math.PI * lat2/180;
  const radlon1 = Math.PI * lon1/180;
  const radlon2 = Math.PI * lon2/180;
  const theta = lon1-lon2;
  const radtheta = Math.PI * theta/180;
  let dist = Math.sin(radlat1) * Math.sin(radlat2) + Math.cos(radlat1) * Math.cos(radlat2) * Math.cos(radtheta);
  dist = Math.acos(dist);
  dist = dist * 180/Math.PI;
  dist = dist * 60 * 1.1515;
  if (unit === "K") { dist = dist * 1.609344 }
  if (unit === "N") { dist = dist * 0.8684 }
  return dist;
}

function abreDialogoUploadArquivoAssinado() {

function carregarArquivo() {
	const fileInput = document.getElementById('fileupload');

	if (fileInput.files.length > 0) {
		const file = fileInput.files[0];
		const reader = new FileReader();

		reader.onload = function(event) {
			const conteudo = event.target.result;
			let xml = stringToXml(conteudo.replace(/[\t\n]/g, ''));
			let placemarks = xml.getElementsByTagName("Placemark");
			for(let i=0; placemarks.length > i; i++){
				let coordenadasArea = placemarks[i].getElementsByTagName("coordinates");
				let name = placemarks[i].getElementsByTagName("name");
				for(let j=0; coordenadasArea.length > j; j++){
					let descricao = name[j] ? name[j].textContent : name[0].textContent;
					preencheCoordenadas(coordenadasArea[j]);
					$("#Descricao").val(descricao);
					$("#salvar").click()
				}
			}
			$("#Coordenada1").val("");
		};

		// Lê o conteúdo do arquivo como texto
		reader.readAsText(file);
	}
}

function stringToXml(conteudo){
	let parser = new DOMParser();
	return parser.parseFromString(conteudo, "text/xml");
}

function preencheCoordenadas(coordenadaArea){
	let pontos = coordenadaArea.textContent.split(" ");
	let coordenadasProcesso =[];
	for(let i=0; pontos.length > i; i++){
		if(pontos[i] != ""){
			let coordenadas = pontos[i].split(',');
			let longitude = coordenadas[0];
			let latitude = coordenadas[1];

			coordenadasProcesso[i]= {"lat":latitude,"lon":longitude};
			prepararCoordenadas(coordenadasProcesso);
		}
	}
}
